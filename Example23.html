Chaining Filters Together

  • {{ name }}

View

<div ng-init="names =['Sai','Kiran','Kokku','Venkata']">
  <ul>
   <li ng-repeat="name in names|exclude: 'Kokku'|sortAscending">
    {{name}}</li>
 </ul>
</div>

Script

                var app = angular.module('myApp', []);
                app.controller('controllerOne', ['$scope', function($scope) {}]);
                app.filter('exclude', function() {
                    return function(input, exclude) {
                        var result = [];
                        for (var i = 0; i < input.length; i++) {
                            if (input[i] !== exclude) {
                                result.push(input[i]);
                            }
                        }
                        return result
                    }

                });
                app.filter('sortAscending', function() {
                    return function(input) {
                        return input.sort();
                    }
                });
