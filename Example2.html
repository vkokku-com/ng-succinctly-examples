Responding to Click Events using Controllers

Hello World

View

<button class="btn btn-info" ng-click="toggle()">Click Here</button>

<h3 ng-hide='visible'>Hello World</h3>

Script

var app = angular.module('myApp', []);

app.controller('controllerOne', ['$scope', function($scope) {
 $scope.visible = true;
 $scope.toggle = function() {
   $scope.visible = !$scope.visible;
 };
}]);
